试题T-2-27 51单片机实现不同按键按下控制点亮不同LED

#include <regx51.h>
sbit LED1=P2^0;
sbit LED2=P2^1;
sbit LED3=P2^2;
sbit SW1=P3^0;
sbit SW2=P3^1;
sbit SW3=P3^2;

void delay(int ms)
{
	int i,j;
	for(i=ms;i>0;i--)
		for(j=110;j>0;j--);
}

void main()
{
	while(1)
	{
		if(SW1==0)
		{
			LED1=0;
			delay(300);
		}else{
			LED1=1;
			delay(300);
		}
		if(SW2==0)
		{
			LED2=0;
			delay(300);
		}else{
			LED2=1;
			delay(300);
		}
		if(SW3==0)
		{
			LED3=0;
			delay(300);
		}else{
			LED3=1;
			delay(300);
		}
	}
}
