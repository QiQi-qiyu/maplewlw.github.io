试题T-2-14 51单片机控制LED灯闪烁

#include <regx51.h>

sbit LED=P2^0;
sbit k1=P1^0;

void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	P2=0XFF;
	P1=0XFF;
	while(1)
	{
		if(k1==0)
		{
			while(k1==0);
			while(1)
			{
				LED=0;
				delay(500);
				LED=1;
				delay(500);
				if(k1==0)
				{
					delay(20);
					while(k1==0);
					break;
				}
			}
		}
	}
}
