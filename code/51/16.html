试题T-2-16 51单片机控制LED灯逐次开关

#include <regx51.h>
sbit LED=P2^0;
sbit LED1=P2^1;
sbit k1=P1^0;

void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	int button=0,state=0;
	P2=P1=0XFF;
	while(1)
	{
		if(k1==0)
		{
			delay(20);
			if(k1==0)
			{
				if(!button)
				{
					button=1;
					state=!state;
				}
				if(state==1)
				{
					LED=0;
					LED1=1;
				}
				else
				{
					LED=1;
					LED1=0;
				}
			}
		}
		else button=0;
	}
}
