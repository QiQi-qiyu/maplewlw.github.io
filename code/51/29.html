#include <regx51.h>
sbit LED1=P2^0;
sbit LED2=P2^1;
sbit LED3=P2^2;
sbit LED4=P2^3;
sbit LED5=P2^4;
sbit LED6=P2^5;
sbit LED7=P2^6;
sbit LED8=P2^7;
sbit SW1=P3^0;
sbit SW2=P3^1;
sbit SW3=P3^2;
sbit SW4=P3^3;

void delay(int ms)
{
	int i,j;
	for(i=ms;i>0;i--)
		for(j=110;j>0;j--);
}

void main()
{
	while(1)
	{
		if(SW1==0)
		{
			//从左至右
			P2=0XFE; 
			delay(300);
			P2=0XFD;
			delay(300);
			P2=0XFB;
			delay(300);
			P2=0XF7;
			delay(300);
			P2=0XEF;
			delay(300);
			P2=0XDF;
			delay(300);
			P2=0XBF;
			delay(300);
			P2=0X7F;
			delay(300);
		}else{
		P2=0XFF;
		delay(300);
		}
		if(SW2==0)
		{//从右至左
			P2=0X7F;
			delay(300);
			P2=0XBF;
			delay(300);
			P2=0XDF;
			delay(300);
			P2=0XEF;
			delay(300);
			P2=0XF7;
			delay(300);
			P2=0XF7;
			delay(300);
			P2=0XFB;
			delay(300);
			P2=0XFD;
			delay(300);
			P2=0XFE;
			delay(300);
		}else{
		P2=0xff;	
		}
		if(SW3==0)
		{//从中间至两头
			P2=0XE7;
			delay(300);
			P2=0XDB;
			delay(300);
			P2=0XBD;
			delay(300);
			P2=0X7E;
			delay(300);
		}else{
			P2=0XFF;
			delay(300);
		}
		if(SW4==0)
		{//从两头至中间
			P2=0X7E;
			delay(300);
			P2=0XBD;
			delay(300);
			P2=0XDB;
			delay(300);
			P2=0XE7;
			delay(300);
		}else{
			P2=0xff;
			delay(300);
		}
	}
}
