试题T-2-12 51单片机控制LED灯轮流闪3次

#include <regx51.h>
sbit LED1=P2^0;
sbit LED2=P2^1;

void delay(unsigned int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	int i;
	P2=0xff;
	for(i=0;i<3;i++)
		{
			LED1=0;
			delay(500);
			LED1=1;
			delay(500);
			LED2=0;
			delay(500);
			LED2=1;
			delay(500);
		}
	while(1);
}
