试题T-2-33 51单片机实现按键控制LED灯左移

#include <regx51.h>
sbit Key=P3^0;

void delay(int ms)
{
	int i,j;
	for(i=ms;i>0;i--)
		for(j=110;j>0;j--);
}

void main()
{
	int i;
		if(Key==0)
		{
			delay(10);
			if(Key==0)
				for(i=0;i<8;i++)
				{
					P2=~(0x80>>i);
					delay(500);
			  }
		}else{
					P2=0XFF;
		}
}
