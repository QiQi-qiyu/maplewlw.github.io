试题T-2-10 51单片机控制多个LED 灯交替闪烁

#include <regx51.h>
sbit LED1=P2^0;
sbit LED2=P2^1;
sbit LED3=P2^2;
sbit LED4=P2^3;
sbit LED5=P2^4;
void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	P2=0XFF;
	while(1)
	{
		LED1=LED3=LED5=0;
		delay(500);
		LED1=LED3=LED5=1;
		delay(500);
		
		LED2=LED4=0;
		delay(500);
		LED2=LED4=0;
		
		P2=0XFF;
	}
	
}
