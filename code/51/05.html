试题T-2-05 51单片机控制 LED 灯从左至右流水灯闪烁

#include <regx51.h>
sbit LED1=P2^0;
sbit LED2=P2^1;
sbit LED3=P2^2;
sbit LED4=P2^3;
sbit LED5=P2^4;

void delay(unsigned int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	LED1=LED2=LED3=LED4=LED5=1;
	while(1)
	{
		LED1=0;
		delay(500);
		LED1=1;
		delay(500);
		
		LED2=0;
		delay(500);
		LED2=1;
		delay(500);
		
		LED3=0;
		delay(500);
		LED3=1;
		delay(500);
		
		LED4=0;
		delay(500);
		LED4=1;
		delay(500);
		
		LED5=0;
		delay(500);
		LED5=1;
		delay(500);
	}
	
}
