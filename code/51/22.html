试题T-2-22 51单片机按键控制LED灯的闪烁与熄灭

#include <regx51.h>
sbit k1=P1^0;
sbit LED=P2^0;

void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

void f_led()
{
	LED=0;
	delay(500);
	LED=1;
	delay(500);
}

int main()
{
	int t=0;
	P2=P1=0XFF;
	while(1)
	{
		if(k1==0)
		{
			delay(20);
			while(k1==0);
			delay(20);
			t++;
		}
		if(t==1)
		{
			f_led();
		}
		else if(t==2)
		{
			P2=0XFF;
			t=0;
		}
	}
		
	
}
