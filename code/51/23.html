试题T-2-23 51单片机按键控制两个LED灯交替闪烁与熄灭

#include <regx51.h>
sbit k1=P1^0;
sbit LED1=P2^0;
sbit LED2=P2^1;

void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
		for(j=110;j>0;j--);
}

int main()
{
	P2=P1=0XFF;
	while(1)
	{
		if(k1==0)
		{
			while(k1==1)
			{
				P2=0XFF;
			}
			LED1=0;LED2=1;
			delay(500);
			while(k1==1)
			{
				P2=0XFF;
			}
			LED1=1;LED2=0;
			delay(500);
		}
		else if(k1==1)
		{
			P2=0XFF;
		}
	}
}
