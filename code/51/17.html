试题T-2-17 51单片机控制流水灯

#include <regx51.h>
sbit k1=P1^0;
int a[5]={0XFE,0XFD,0XFB,0XF7,0XEF};
void delay(int xms)
{
	int i,j;
	for(i=xms;i>0;i--)
			for(j=110;j>0;j--);
}

int main()
{
	int i,t=0,index=0;
	P2=P1=0XFF;
	while(1)
	{
		if(k1==0)
		{
			delay(20);
			while(k1==0);
			delay(20);
			t++;
		}
		if(t==1)
		{
			while(t==1)
			{
				P2=a[index];
				delay(500);
				index=(index+1)%5;
				if(k1==0)
				{
					delay(20);
					while(k1==0);
					delay(20);
					t++;
				}
			}
		}
		else if(t==2)
		{
			P2=a[index-1];
			while(t==2)
			{
				if(k1==0)
				{
					delay(20);
					while(k1==0);
					delay(20);
					t=1;
				}
			}
		}
	}
	return 0;
}
