试题T-1-17按键开关控制LED灯从中间至两头闪烁

#include <iocc2530.h>
#define LED1 P1_1
#define LED2 P1_0
#define LED3 P1_4
#define LED4 P1_3
#define SW P1_2
int state=0;

void delay(int xms)
{
  int i,j;
  for(i=xms;i>0;i--)
     for(j=110;j>0;j--);
}

int main()
{
  P1DIR|=0x1B;
  P1DIR&=~0x04;
  LED1=LED2=LED3=LED4=0;
  while(1)
  {
    if(SW==0)
    {
      delay(20);
      while(SW==0);
      delay(20);
      state++;
    }
    if(state==1)
    {
     LED2=LED3=1;
      delay(500);
     LED2=LED3=0;
      delay(500);
     
      LED1=LED4=1;
      delay(500);
     LED1=LED4=0;
      delay(500);
    }
  }
}
