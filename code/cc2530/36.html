试题T-1-36 利用定时器实现2个LED灯同时闪烁，周期１秒

#include <iocc2530.h>
#define LED1 P1_0
#define LED2 P1_1

void main()
{
  char count;
  
  P1DIR |=0X03;
  LED1=LED2=0;
  T1CTL=0X0D;
  
  while(1)
  {
    if(IRCON>0)
    {
      IRCON=0;
      if(++count==2)
      {
        count=0;
        LED1=!LED1;
        LED2=!LED2;
      }
    }
  }
}
