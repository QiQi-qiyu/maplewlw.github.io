试题T-1-22 按键按下时2个LED灯同时闪烁，松开时暂停

#include <iocc2530.h>
#define LED1 P1_0
#define LED2 P1_1
#define SW P1_2

void delay(int ms)
{
  int i,j;
  for(i=ms;i>0;i--)
    for(j=500;j>0;j--);
}

void main()
{
  P1DIR |=0X03;
  P1DIR &=~0X04;
  P1=0X00;
  while(1)
  {
    if(SW==0)
    {
      delay(10);
        if(SW==0)
        {
          LED1=1;
          LED2=1;
          delay(300);
          
          LED1=0;
          LED2=0;
          delay(300);
        }
        else
        {
          P1=0X00;
        }
    }
  }
}
