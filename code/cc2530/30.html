试题T-1-30 按键开关控制LED灯跑马灯模拟八进制数据表示

#include <iocc2530.h>
#define LED1 P1_1
#define LED2 P1_0
#define LED3 P1_4

void dealy(int ms)
{
  int i,j;
  for(i=ms;i>0;i--)
    for(j=500;j>0;j--);
}

void main()
{
  
  P1DIR|=0X1B; //打开p1到p4
  
  while(1)
  {
   //0
   P1=0x00;  
   dealy(500);
   //1
   LED1=1;
   LED2=LED3=0;
   dealy(500);
   //2
   LED2=1;
   LED1=LED3=0;
   dealy(500);
   //3
   LED1=LED2=1;
   LED3=0;
   dealy(500);
   //4
   LED3=1;
   LED1=LED2=0;
   dealy(500);
   
   //5
   LED1=LED3=1;
   LED2=0;
   dealy(500);
   
   //6
   LED2=LED3=1;
   LED1=0;
   dealy(500);
   
   //7
   LED1=LED2=LED3=1;
   dealy(500);
  }
}
