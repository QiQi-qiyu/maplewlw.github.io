试题 T-1-19 按键按下时LED1亮，松开时LED1灯灭

#include <iocc2530.h>
#define LED1 P1_1
#define SW P1_2
int state=0;

void delay(int xms)
{
  int i,j;
  for(i=xms;i>0;i--)
     for(j=110;j>0;j--);
}

int main()
{
  P1DIR|=0x02;
  P1DIR&=~0x04;
  LED1=0;
  while(1)
  {
    if(SW==0)
    {
     LED1=1;
    }
    else LED1=0;
  }
}
