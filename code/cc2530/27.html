试题T-1-27 按键开关控制LED灯实现跑马灯启停功能

#include <iocc2530.h>
#define LED1 P1_1
#define LED2 P1_0
#define LED3 P1_4
#define LED4 P1_3

#define SW P1_2

void delay(int xms)
{
  int i,j;
  for(i=xms;i>0;i--)
    for(j=110;j>0;j--);
}

int main()
{
  P1DIR|=0x1B;
  P1DIR&=~0x04;
  
  int count=0;
  int state=0;
  int currentLED=1;

  LED1=LED2=LED3=LED4=0;

  while(1)
  {
    if(SW==0)
    {
      while(SW==0);
      state++;
    }
    if(state==1)
    {
      if(currentLED==1) {LED1=1;LED2=LED3=LED4=0;  delay(1500);  currentLED++;}
      
      else if(currentLED==2){LED2=1;LED1=LED3=LED4=0;  delay(1500);  currentLED++;}
      
      else if(currentLED==3){LED3=1;LED1=LED2=LED4=0;  delay(1500);  currentLED++;}
      
      else if(currentLED==4){LED4=1;LED1=LED3=LED2=0;  delay(1500);  currentLED++;}
        
      if(currentLED > 4){currentLED=1;}
    }  
    
    else if(state == 2)
    {
      count=currentLED-1;
      if(count==1){LED1=1;LED2=LED3=LED4=0;}
      else if(count==2) {LED2=1;LED1=LED3=LED4=0;}
      else if(count==3) {LED3=1;LED1=LED2=LED4=0;}
      else if(count==4) {LED4=1;LED1=LED2=LED3=0;}
      state=0;
    }
    else if(state>2)
    {
      state=0;
      currentLED=1;
    }
  }
}
