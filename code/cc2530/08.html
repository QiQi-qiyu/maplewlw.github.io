试题T-1-08 CC2530程序控制LED灯从两头至中间闪烁

#include <iocc2530.h>
#define LED1 P1_0
#define LED2 P1_1
#define LED3 P1_3
#define LED4 P1_4

void dealy(int ms)
{
  int i,j;
  for(i=ms;i>0;i--)
    for(j=500;j>0;j--);
}

void main()
{
  P1DIR|=0X1B; //打开p1到p4
  
  while(1)
  {
    LED1=0;  //从两头至中间
    LED4=0;
    LED2=1;
    LED3=1;
    dealy(300);
    
    LED1=1;
    LED4=1;
    LED2=0;
    LED3=0;
    dealy(300);
  }
}
