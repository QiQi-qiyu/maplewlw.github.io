试题T-1-12 按键开关控制LED1灯闪烁

#include <iocc2530.h>
#define LED1 P1_0
#define LED2 P1_1
#define SW P1_2

int stata=0;

void delay(int xms)
{
  int i,j;
  for(i=xms;i>0;i--)
     for(j=110;j>0;j--);
}

void main()
{
  P1DIR|=0x01;
  P1DIR&=~0x04;

  LED1=0;
  while(1)
  {
    if(SW==0)
    {
      delay(20);
      while(SW==0);
      delay(20);
      stata++;
    }
    if(stata==1)
    {
      LED1=1;
      delay(1000);
      LED1=0;
      delay(1000);
    }
    if(stata==2)
    {
      LED1=0;
      stata=0;
    }
  }
}
