试题T-1-23 按键按下时控制LED灯从左至右闪烁，松开时停止

#include <iocc2530.h>
#define SW P1_2

unsigned char Led_Left[]={0x08,0x10,0x01,0x02};
void delay(int ms)
{
  int i,j;
  for(i=ms;i>0;i--)
    for(j=500;j>0;j--);
}

void main()
{
  int i;
  P1DIR |=0X1b;
  P1DIR &=  ~0X04;
  P1=0X00;
  while(1)
  {
    if(SW==0)
    {
      delay(10);
        if(SW==0)
        {
          for(i=0;i<4;i++)
          {
            P1=Led_Left[i];
            delay(300);
             P1=0X00;
          }
        }else
        {
          P1=0X00;
        }
    }
  }
}
