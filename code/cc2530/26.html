试题T-1-26 按键按下时控制LED灯从两头至中间闪烁，松开时暂停

#include <iocc2530.h>
#define SW P1_2

void delay(int ms)
{
  int i,j;
  for(i=ms;i>0;i--)
    for(j=500;j>0;j--);
}

void main()
{
  P1DIR |=0X1B;
  P1DIR &=  ~0X04;
  P1=0X00;
  while(1)
  {
    if(SW==0)
    {
      delay(10);
        if(SW==0)
        {
          P1=0X0A;
          delay(300);
           P1=0X00;
           
          P1=0X11;
          delay(300);
          P1=0X00;
          
        }else
        {
          P1=0X00;
          P0=0X00;
        }
    }
  }
}
