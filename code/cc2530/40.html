试题T-1-40 利用按键触发定时器实现1个LED灯闪烁，周期2秒

#include <iocc2530.h>
#define LED1 P1_0
#define LED2 P1_1
#define SW P1_2

int state=0;

void main()
{
  char count;
  
  P1DIR |=0X03;
  LED1=LED2=0;
  T1CTL=0X0D;
  
  while(1)
  {
    if(SW==0)
    {
      while(SW==0);
      state++;
    }
    if(state==1)
    {
      if(IRCON>0)
      {
        IRCON=0;
        if(++count==4)
        {
          count=0;
          LED1=!LED1;
          LED2=!LED2;
        }
      }
    }
    
  }
}
