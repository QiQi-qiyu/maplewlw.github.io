试题T-1-28 按键按下时控制LED灯实现跑马灯功能，松开时暂停

#include <iocc2530.h>
#define LED1 P1_1
#define LED2 P1_0
#define LED3 P1_4
#define LED4 P1_3
#define SW P1_2

void delay(int xms)
{
  int i,j;
  for(i=xms;i>0;i--)
     for(j=110;j>0;j--);
}

int main()
{
  P1DIR|=0x1B;
  P1DIR&=~0x04;
  P1=0X00;
  while(1)
  {
    if(SW==0)
    {
      LED1=1;
      LED2=LED3=LED4=0;
      delay(1000);
      
      LED2=1;
      LED1=LED3=LED4=0;
      delay(1000);
       
      LED3=1;
      LED1=LED2=LED4=0;
      delay(1000);
       
      LED4=1;
      LED2=LED3=LED1=0;
      delay(1000);
    }
    else 
    {
      P1=0X00;
    }
  }
}
